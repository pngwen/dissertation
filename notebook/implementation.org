* outer(a,b)
dims = a.dim U b.dim
fill = a.fill * b.fill
foreach non-fill index(ia) in a
  foreach non-fill index(ib) in b
    set(ia U ib, a[ia]*a[ib])
  end foreach
end foreach
return * this

* hadamard(a,b)
  fill = a.fill * b.fill
  multList = a.nonfill() U b.nonfill()
  foreach idx in multList
    set(idx, a[idx]*b[idx])
  end foreach
  return * this
